import java.util.*;


public class Solution {
    /**
     * 代码中的类名、方法名、参数名已经指定，请勿修改，直接返回方法规定的值即可
     *
     *
     * @param n int整型
     * @return int整型
     */
    public int Fibonacci (int n) {
        // write code here
        if (n <= 1) return n;
        int cur = 0;
        int a = 0;
        int b = 1;

        for (int i = 2; i <= n; i++) {
            cur = a + b;
            a = b;
            b = cur;
        }
        return cur;
    }
}
